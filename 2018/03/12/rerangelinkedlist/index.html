struct node{
    int add;
    int data;

    int temp[100000][2];
    struct node in[100000];
    scanf("%d%d",&head,&n);
    int i;
    for(i=0;i<n;i++){
        int temp1,temp2,temp3;

        temp[temp1][0]=temp2;
        temp[temp1][1]=temp3;
    }

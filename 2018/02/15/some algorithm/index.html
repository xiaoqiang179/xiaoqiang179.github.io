快速幂：

double poww(int a,int b){
    double ans=1,base=a;
    while(b!=0){
        if((b&1)!=0){
            ans*=base;
        }
        base*=base;
        b>>=1;
    }
    return ans;
}

快速幂取模

long long quick(long long a,long long b，int c){
    long long ans=1;   //记录结果
    a=a%c;   //预处理，使得a处于c的数据范围之下
    while(b!=0){
        if(b&1) ans=(ansa)%c;   //如果b的二进制位不是0，那么我们的结果是要参与运算的
        b>>=1;    //二进制的移位操作，相当于每次除以2，用二进制看，就是我们不断的遍历b的二进制位
        a=(aa)%c;   //不断的加倍
    }
    return ans;
}

判断素数

int pdprime(long long n){       //判断素数，1是，0不是
    int flag=1;
    for(int i=2;i<=sqrt(n);i++){
        if((long long)n%i==0){
            flag=0;
            break;
        }
    }
    if(n==1){
        flag=0;
    }
    return flag;
}

求斐波那契额数列

递归法
int f( int n ){
    if(n==0){
        return 0;
    }     
    else if(n==1){
        return 1;
    }                  
    else{
        return f(n-2)+f(n-1);
    }
}
普通方法  //我自己想的😜
int f(int n){
    int i,a[3],i2,i3;
    if(n==0){
        return 0;
    }
    else if(n==1){
        return 1;
    }
    else{
        i2=2;
        a[0]=0;
        a[1]=1;
        for(i=2;i<=n;i++){
            if(i2==2){
                a[i2]=a[1]+a[0];
            }
            else if(i2==1){
                a[i2]=a[0]+a[2];
            }
            else{
                a[i2]=a[1]+a[2];
            }
            i3=i2;
            i2++;
            if(i2==3){
                i2=0;
            }
         }
        return a[i3];
    }
}

    int b=a*1000;

        b+=10;
        b=b/10;
